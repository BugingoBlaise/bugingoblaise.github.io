// Example of a scalable service
class UserService {
    constructor() {
        this.cache = new Cache();
        this.db = new Database();
    }

    async getUser(id) {
        // Try cache first
        let user = await this.cache.get(id);
        if (!user) {
            // Fall back to database
            user = await this.db.findUser(id);
            // Update cache
            await this.cache.set(id, user);
        }
        return user;
    }
}
